# DexGuard's default settings are fine for this sample application.

# Display some more statistics about the processed code.
-verbose

# Encrypt our secret class.
-encryptclasses com.example.SecretClass

# We'll also encrypt the message strings, providing another layer of obfucation.
-encryptstrings "Hello *!"

# For this example, we want to avoid the trivial test method being inlined
# from the secret class, before the class is encrypted.
-keep,allowshrinking,allowobfuscation class com.example.SecretClass {
    <methods>;
}
